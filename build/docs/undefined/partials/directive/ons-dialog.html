ons-dialog

# 

<script>
  ons.ready(function() {
    ons.createDialog('dialog.html').then(function(dialog) {
      dialog.show();
    });
  });
</script>

<script type="text/ons-template" id="dialog.html">
  <ons-dialog cancelable>
    ...
  </ons-dialog>
</script>
